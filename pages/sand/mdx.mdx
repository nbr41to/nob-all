export const mdxComponents = {
  h1: (props: ComponentProps<'h1'>) => (
    <h1 className="text-3xl" {...props}>
      {props.children}
    </h1>
  ),
  pre: (props: ComponentProps<'pre'>) => (
    <pre className="rounded bg-gray-300 p-1" {...props}>
      {props.children}
    </pre>
  ),
};
